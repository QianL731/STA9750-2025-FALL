Introduction

In August 2025, President Donald Trump fired Dr. Erika McEntarfer, Commissioner of Labor Statistics, claiming that the jobs numbers were “rigged.” The jobs number refers to the Current Employment Statistics (CES) total nonfarm payroll, a monthly estimate that often moves markets and shapes political narratives.

This mini-project investigates whether revisions to those jobs numbers are:

  • unusually large,

  • systematically biased, or

  • consistent with the normal statistical process of updating estimates.

The analysis follows the instructions:

  • Task 1: Use httr2 + rvest to download CES total nonfarm employment (seasonally adjusted) from BLS’s Data Finder (PDQ).

  • Task 2: Use httr2 + rvest to scrape CES revision tables from cesnaicsrev.htm.

  • Task 3: Join and explore the data with summary stats and visualizations.

  • Task 4: Use infer to run at least two hypothesis tests on revision patterns.

  • Task 5: Write data-based fact checks of public claims about CES revisions.

Because my environment sometimes cannot reach BLS servers, I include automatic fallbacks: if scraping fails, the code will look for local cached CSV files; if those are absent, it uses a simulated dataset with the same structure so the document can still render.

# Helper: fake CES levels if everything else fails

generate_fake_ces_levels <- function() {
dates <- seq(ymd("1979-01-01"), ymd("2025-06-01"), by = "month")
n <- length(dates)

base_level <- 88e6
trend      <- seq(0, 70e6, length.out = n)  # gradual growth

# Simple seasonal pattern

seasonal_pattern <- c(-200000,  0, 200000, 300000,
400000, 200000,      0, -200000,
-300000,-200000,      0, 100000)
seasonal <- rep(seasonal_pattern, length.out = n)

tibble(
date  = dates,
level = base_level + trend + seasonal
)
}

# Helper: fake revisions if scraping & cached fail

generate_fake_ces_revisions <- function(ces_level) {
ces_level |>
arrange(date) |>
mutate(
change_jobs      = level - lag(level),
change_thousands = change_jobs / 1000
) |>
filter(!is.na(change_thousands)) |>
mutate(
idx      = row_number(),
original = round(change_thousands + sin(idx) * 10),
final    = round(change_thousands + sin(idx + 0.5) * 10),
revision = final - original
) |>
select(date, original, final, revision)
}

Task 1 – CES Total Nonfarm Payroll (1979–2025)

We first try to scrape the PDQ HTML page using httr2 and rvest, as required. If that fails (e.g., due to “Bad IPv6 address” / 403), we fall back to:

  • A local cached file ces_level_cached.csv (if you have one),

  • Otherwise, a simulated CES series.

get_ces_levels <- function() {

# 1) Try scraping from BLS PDQ (Top Picks)

ces_scraped <- tryCatch({

ces_req <- request("https://data.bls.gov/pdq/SurveyOutputServlet") |>
  req_method("POST") |>
  req_headers(
    "User-Agent" = "Mozilla/5.0",
    "Origin"     = "https://data.bls.gov",
    "Referer"    = "https://data.bls.gov/toppicks?survey=ce"
  ) |>
  req_body_form(
    survey      = "ce",
    series_id   = "CEU0000000001",  # Total nonfarm, all employees, SA
    from_year   = "1979",
    to_year     = "2025",
    output_type = "0",              # levels
    time        = "M",              # monthly
    delimiter   = "comma"
  )

ces_html <- ces_req |>
  req_perform() |>
  resp_body_html()

ces_table <- ces_html |>
  html_elements("table") |>
  pluck(1) |>
  html_table(fill = TRUE)

ces_clean <- ces_table |>
  clean_names() |>
  rename(year = 1) |>
  filter(year != "", !is.na(year)) |>
  mutate(year = as.integer(year)) |>
  pivot_longer(
    cols = -year,
    names_to  = "month_name",
    values_to = "raw_value"
  ) |>
  mutate(
    month_name = str_to_title(month_name),
    month_name = if_else(month_name == "Sept", "Sep", month_name),
    ym_str     = paste(year, month_name),
    date       = ym(ym_str),
    level      = readr::parse_number(raw_value) * 1000
  ) |>
  select(date, level) |>
  drop_na() |>
  arrange(date) |>
  filter(date <= ymd("2025-06-01"))

message("✔ CES levels successfully scraped from BLS.")
ces_clean

}, error = function(e) {
message("⚠ Failed to scrape CES levels from BLS: ", conditionMessage(e))
NULL
})

if (!is.null(ces_scraped)) return(ces_scraped)

# 2) Try local cache

if (file.exists("ces_level_cached.csv")) {
message("✔ Loading CES levels from ces_level_cached.csv")
return(read_csv("ces_level_cached.csv", show_col_types = FALSE) |>
mutate(date = ymd(date)))
}

# 3) Fallback: simulated data so document still renders

message("⚠ Using simulated CES levels as fallback.")
generate_fake_ces_levels()
}
